int fib(int last){

int previous;

int current;

int aux;

int c;

previous = 0;

current = 1;

c = 0;

for 0:last then

c = c + 1;

if(c <= 1) then

aux = c;

else

aux = previous + current;

previous = current;

current = aux;

end;

output (string) aux;

return 0;

end;

};

int main(){

fib(8);

return 0;
};
